// MODULE: DeepResearchEngine (TTD-DR Integration)

FUNCTION TTD_DR_EXECUTION(user_query):
    ROUND = "PLAN"

    // ===========================
    // 1. PLAN ROUND — Clarify Intent
    // ===========================
    intent = clarify_user_intent(user_query)
    IF intent.confidence < 0.9:
        intent = request_clarification(user_query)
    ENDIF

    research_goal = define_research_goal(intent)
    log("Research goal defined: ", research_goal)

    ROUND = "RESEARCH"

    // Split into subtopics for parallel exploration
    // ===========================
    // 2. RESEARCH ROUND — Research Brief + Decomposition
    // ===========================
    research_brief = build_research_brief(research_goal)
    subtopics = extract_subtopics(research_brief)
    assign_to_agents(subtopics)

    ROUND = "BLUEPRINT"

    // ===========================
    // 3. BLUEPRINT ROUND — Initial Draft
    // ===========================
    draft = generate_initial_draft(research_goal, research_brief)

    ROUND = "DETAIL"

    // ===========================
    // 4. DETAIL ROUND — ReAct + Denoising Loop
    // ===========================
    iteration = 0
    MAX_ITER = 5
    convergence = 0.0

    WHILE (iteration < MAX_ITER) AND (convergence < 0.9):
        iteration += 1
        log("Iteration:", iteration)

        // --- ReAct Loop ---
        evidence_bundle = []
        FOR each topic IN subtopics:
            evidence = worker_agent_research(topic)
            IF evidence.is_valid:
                evidence_bundle.append(evidence)
            ENDIF
        ENDFOR

        // --- Core Denoising Loop ---
        refined_draft = denoise_draft(draft, evidence_bundle)
        refined_draft = fill_information_gaps(refined_draft)
        refined_draft = normalize_citations(refined_draft)

        // --- Supervisor Oversight ---
        supervisor_feedback = supervisor_agent_review(refined_draft)
        IF supervisor_feedback.requires_replan:
            research_brief = update_research_brief(supervisor_feedback)
            subtopics = extract_subtopics(research_brief)
            CONTINUE
        ENDIF

        // --- Red Teaming Phase ---
        attack_report = red_team_attack(refined_draft)
        IF attack_report.flaws_detected:
            refined_draft = fix_flaws(refined_draft, attack_report)
        ENDIF

        // --- Evaluation Phase ---
        eval_score = evaluator_agent_score(refined_draft)
        convergence = compute_convergence(eval_score, previous_score)
        previous_score = eval_score

        log("Eval score:", eval_score, "Convergence:", convergence)
    ENDWHILE

    ROUND = "QA"

    // ===========================
    // 5. QA ROUND — Validation & Safety
    // ===========================
    security_check = run_security_gate(refined_draft)
    privacy_check  = run_privacy_gate(refined_draft)
    consistency_check = verify_logic_consistency(refined_draft)

    IF (NOT security_check.passed) OR (NOT privacy_check.passed):
        RAISE Exception("Compliance Gate Failure")
    ENDIF

    IF consistency_check.score < 0.85:
        refined_draft = reanalyze_inconsistencies(refined_draft)
    ENDIF

    ROUND = "HANDOFF"

    // ===========================
    // 6. HANDOFF ROUND — Final Report
    // ===========================
    final_report = {
        "title": generate_title(research_goal),
        "iterations": iteration,
        "confidence": eval_score,
        "citations": extract_citations(refined_draft),
        "key_findings": summarize_findings(refined_draft),
        "remaining_gaps": detect_remaining_gaps(refined_draft),
        "next_actions": propose_next_steps(refined_draft)
    }

    save_report(final_report, path="/reports/deep_research/")
    audit_log(final_report)

    RETURN final_report
END FUNCTION

// ======================================================
// Supporting Utility Functions
// ======================================================

FUNCTION clarify_user_intent(query):
    result = LLM_classify_and_disambiguate(query)
    RETURN { intent_text: result.text, confidence: result.confidence }
END FUNCTION

FUNCTION generate_initial_draft(goal):
    RETURN LLM_generate_outline(goal)
END FUNCTION

FUNCTION build_research_brief(draft):
    RETURN extract_questions_and_sources(draft)
END FUNCTION

FUNCTION worker_agent_research(topic):
    evidence = fetch_verified_data(topic)
    RETURN validate_evidence(evidence)
END FUNCTION

FUNCTION denoise_draft(draft, evidence):
    RETURN apply_diffusion_refinement(draft, evidence)
END FUNCTION

FUNCTION red_team_attack(draft):
    RETURN analyze_for_bias_and_logic_flaws(draft)
END FUNCTION

FUNCTION evaluator_agent_score(draft):
    factual = measure_factual_accuracy(draft)
    logic   = measure_logical_coherence(draft)
    coverage= measure_topic_coverage(draft)
    RETURN average([factual, logic, coverage])
END FUNCTION

FUNCTION compute_convergence(current, previous):
    RETURN 1 - ABS(current - previous)
END FUNCTION
